\begin{defn}
	A \textit{monoidal structure} on a category $ \mathcal{V}$ is given by a functor $\otimes: \mathcal{V}\times \mathcal{V} \longrightarrow \mathcal{V}$	
	which is associative
	\[
	u \otimes (v\otimes w) \simeq (u\otimes v) \otimes w
	\] 
	and has a unit object $ k$ such that
	\[
	k \otimes v \simeq v \simeq v \otimes k.
	\] 
	It is furthermore \textit{symmetric} if 
	\[
	u \otimes v \simeq v \otimes u.
	\] 
\end{defn}

\begin{defn}
	A monoidal structure is \textit{closed} if there is a functor
	$\mathcal{V}(-,-):\mathcal{V}^{ \textsf{op}} \times \mathcal{V} \longrightarrow \mathcal{V}$
	such that $ \mathcal{V}(a, -)$ is right adjoint to $ a \, \otimes \, -$ for each $ a$ in $ \mathcal{V}$.
	In other words,
	\[
	\mathcal{V}(a  \otimes  -, *) \approx \mathcal{V}(-, \mathcal{V}(a, *)).
	\]
	In particular, there is a natural bijection $\mathcal{V}(a,v) \approx \mathcal{V}(k, \mathcal{V}(a,v)).$\\
	
\end{defn}

Let's look at the data of this adjunction. 
The unit and counit are $\lambda _{u}: u \longrightarrow \mathcal{V}(a,a \otimes u)$ and	$\epsilon _{v}: a \, \otimes \,  \mathcal{V}(a,v) \longrightarrow v$.
The bijections are given by the procedure:

\begin{center}
	$a\, \otimes \, u \xrightarrow{f} v \text{  is sent to  } u \xrightarrow{ \lambda_{u}}  \mathcal{V}(a, a\, \otimes \, u) \xrightarrow{ \mathcal{V}(a,f)} \mathcal{V}(a,v)$\\
	
	$u \xrightarrow{g}  \mathcal{V}(a,v)  \text{  is sent to  } a\, \otimes \, u \xrightarrow{ a \, \otimes \, g} a \, \otimes \mathcal{V}(a,v) \xrightarrow{ \epsilon _{,v}} v$.
\end{center}


Since $ \epsilon $ can be thought of as an evaluation, a form of composition $	\mathcal{V}(a,b) \otimes \mathcal{V}(b,c) \longrightarrow \mathcal{V}(a,c)$
which is calculated at each $ a$ in $ \mathcal{V}$ as:
\begin{center}
	$a \otimes ( \mathcal{V}(a,b) \otimes \mathcal{V}(b,c) ) \xrightarrow{ \simeq} 	(a \otimes  \mathcal{V}(a,b)) \otimes \mathcal{V}(b,c) 
	\xrightarrow{ \epsilon_b \otimes 1 _{ \mathcal{V}(b,c)}} b \otimes \mathcal{V}(b,c) \xrightarrow{ \epsilon'_c} c. $\\
\end{center}
There is also a natural double-dual morphism $u \longrightarrow \mathcal{V}( \mathcal{V}(u,b),b)$ for any fixed object $ b$.\\

\begin{defn}
	For $ \mathcal{V} $-categories $ \textbf{A}$ and $ \textbf{B}$, a \textit{ $ \mathcal{V}$-functor} $ T: \textbf{A} \longrightarrow  \textbf{B}$ consists of a function 
	\[
	T:\textsf{ob} \textbf{A} \longrightarrow \textsf{ ob} \textbf{B}	
	\] 
	together with, for each pair $ a,b \in \textsf{ob} \textbf{A}$, a map 
	\[
	T _{ab}: \textbf{A}(a,b) \longrightarrow \textbf{B}(Ta,Tb)  
	\] 
	subject to the compatibility with composition and the identities.
\end{defn}

\begin{defn}
	A functor $ \mathcal{V} \xrightarrow{\phi} \mathcal{W}$ between monoidal structures is \textit{closed} if it preserves the monoidal structure of $ \mathcal{V}$ (the identity, $\otimes$).
\end{defn}

\begin{thm} Our results:
	\begin{enumerate}
		\item If $\mathcal{V} \xrightarrow{\phi} \mathcal{W}$ is a closed functor and \textbf{A} a $ \mathcal{V}$-category, then $ (\phi \textbf{A})(a,b) = \phi( \textbf{A}(a,b))$ defines a $\mathcal{W}$-category $\phi \textbf{A}$ with the same objects as \textbf{A}.
		\item If $ \textbf{A} \xrightarrow{T} \textbf{B}$ is a $\mathcal{V}$-functor, then we obtain a $ \mathcal{W}$ functor $\phi \textbf{A} \xrightarrow{\phi T} \phi \textbf{B}$.
		\item If $ \mathcal{V}$ is small and $ \mathcal{V} \xrightarrow{\phi} \mathcal{W}$ is a closed functor, then $ \phi \mathcal{V}$ is a $\mathcal{W}$-category. 
	\end{enumerate}
\end{thm}
